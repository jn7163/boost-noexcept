namespace boost { namespace noexcept_ {

    template <class R>
    struct throw_return {

        static R value() noexcept;

    }

    class throw_ {

        throw_( throw_ const & )=delete;
        throw_ & operator=( throw_ const & )=delete;

        public:

        throw_() noexcept { }

        template <class E>
        throw_( E && e ) noexcept;

        template <class R>
        operator R() noexcept;
        
    };

    bool has_current_error() noexcept;

} }
