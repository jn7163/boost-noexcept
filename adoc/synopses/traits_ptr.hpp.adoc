namespace boost { namespace noexcept_ {

    template <class> struct result_traits;

    template <class T>
    struct result_traits<T *> {

        static bool succeeded( T * x ) noexcept {
            return bool(x);
        }

        static T * error_result() noexcept {
            return nullptr;
        }
    };

} }
